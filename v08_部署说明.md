### 2. 导航 JavaScript 函数
```javascript
// 返回主宇宙
function goToMainUniverse() {
    playSound('sceneChange');
    renderScene("start");
    updateUniverseSelector("start");
}

// 切换宇宙
function switchUniverse(sceneId) {
    playSound('sceneChange');
    renderScene(sceneId);
    updateUniverseSelector(sceneId);
}

// 更新宇宙选择器状态
function updateUniverseSelector(sceneId) {
    const selector = document.getElementById('universe-select');
    if (selector) {
        selector.value = sceneId;
    }
}
```
